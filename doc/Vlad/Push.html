def push_cleanup
  [ "rm -vrf /tmp/#{application}-*.tgz",
    [ "if [ -e #{repository} ]",
      "then rm -rf #{repository}",
      "fi"
    ].join("; ")
  ].join(" && ")
end

def push_extract
  [ "if [ -e #{repository} ]; then rm -rf #{repository}; fi",
    "mkdir -p #{repository}",
    "cd #{repository}",
    "tar -xzf #{extract_dir}/#{extract_file}"
  ].join(" && ")
end
