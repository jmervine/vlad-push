  def push(host)
    [ "#{push_scp} #{ssh_flags}",
      "/tmp/#{application}-#{release_name}.tgz",
      "#{host}:/tmp/#{application}-#{release_name}.tgz"
    ].join(" ")

  def push_cleanup
    [ "rm -vrf /tmp/#{application}-*.tgz",
      [ "if [ -e #{repository} ]",
        "then rm -rf #{repository}",
        "fi"
      ].join("; ")
    ].join(" && ")
  end
